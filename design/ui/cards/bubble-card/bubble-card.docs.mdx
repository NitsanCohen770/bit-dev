import { BubbleCard } from './bubble-card';
import { Button } from '@teambit/design.ui.buttons.button';

### Component usage


```tsx
<BubbleCard style={{ width: 170 }}>
  <p>Getting Started</p> 
</BubbleCard>
```

### Rendering a button in a Bubble card

```js live
<BubbleCard style={{ width: 170 }}>
  <Button>Getting Started</Button>
</BubbleCard>
```
