---
description: "Isomorphic link provder."
labels: ["react", "routing", "context"]
---

Sets the underlying implementation of links in the project.

As simple as:

```tsx
import { link, useLocation } from "react-router-dom";
import { RoutingProvider } from "./router-context-provider";

const routingSystem = {
  Link: ({ to, ...rest }) => <Link {...rest} to={href} />,
  useLocation: useLocation,
};

// ...

return <RoutingProvider value={{ routingSystem }}>{children}</RoutingProvider>;
```
